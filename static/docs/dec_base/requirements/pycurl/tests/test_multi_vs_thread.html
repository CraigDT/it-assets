class TestPool(Thread):

      		
    

      
        

          #
        

        
      		
      
        
    def __init__(self, lock, pool):
        Thread.__init__(self)
        self.lock = lock
        self.pool = pool

      		
    

      
        

          #
        

        
      		
      
        
    def run(self):
        while 1:
            self.lock.acquire()
            c = None
            if self.pool:
                c = self.pool.pop()
            self.lock.release()
            if c is None:
                break
            c.perform()
