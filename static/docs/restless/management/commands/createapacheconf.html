import os, shutil, subprocess

from django.core.management.base import BaseCommand, CommandError
from django.conf import settings
# 
class Command(BaseCommand):
    help = 'Deploys configuration (apache) and clean static dir'
# 
    def handle(self, *args, **options):
        if not hasattr(settings, "APACHE_EXTRA"):
            settings.APACHE_EXTRA = ""
        apacheconf = '''
Listen {0.SITE_PORT}
<VirtualHost *:{0.SITE_PORT}>{0.APACHE_EXTRA}
    Alias /static {0.STATIC_ROOT}
    WSGIScriptAlias / {0.SITE_ROOT}/apache.wsgi
</VirtualHost>
        '''.format(settings).strip()
        with open(os.path.join(args[0], "{0}.conf".format(os.path.basename(settings.SITE_ROOT))), "w") as apacheconffile:
            apacheconffile.write(apacheconf)
        shutil.rmtree(settings.STATIC_ROOT)
        subprocess.call("service apache2 restart".format(os.path.basename(settings.SITE_ROOT)), shell=True)
        print("Application now being served at http://{0}:{1}/\n".format(settings.HOSTNAME, settings.SITE_PORT))
        return 0
